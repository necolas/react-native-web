function encodeGroupRule(group) {

}
 
function decodeGroupRule(cssRule) {

function getOrderedGroups(obj: { [key: number]: any }) {
  return Object.keys(obj)
    .map(Number)
    .sort((a, b) => (a > b ? 1 : -1));
